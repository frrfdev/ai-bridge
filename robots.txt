# GET /breeds
# Description: Retrieve a list of all dog breeds
# Parameters: None
# Response: JSON object with breeds array containing id, name, description, life_span, etc.
# Example:
#   fetch('https://dogapi.dog/api/v2/breeds')
#   → { "data": [ { "id": "68f511c2...", "attributes": { "name": "Golden Retriever", ... } } ] }

# GET /breeds/{id}
# Description: Retrieve detailed information about a specific breed
# Parameters:
#   - id (path parameter): UUID string of the breed
#     Obtain valid IDs from the /breeds endpoint first
# Response: JSON object with single breed details
# Example:
#   fetch('https://dogapi.dog/api/v2/breeds/68f511c2-8d38-4dc1-932a-2537e71c70f5')
#   → { "data": { "id": "68f511c2...", "attributes": { ... } } }

# GET /facts
# Description: Retrieve random dog facts
# Parameters:
#   - limit (optional query parameter): Number of facts to return
#     Default: varies by API
# Response: JSON object with facts array
# Example:
#   fetch('https://dogapi.dog/api/v2/facts')
#   → { "data": [ { "id": "1", "attributes": { "body": "Dogs have a sense of time." } } ] }
#   fetch('https://dogapi.dog/api/v2/facts?limit=5')
#   → { "data": [ ...5 facts... ] }
